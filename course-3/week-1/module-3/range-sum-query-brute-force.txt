
//https://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/Y 
// Y. Range sum query

#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int n, q;
    cin >> n >> q;
    
    vector<long int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    for (int i = 0; i < q; i++) {
        int l, r; // 1 based index
        cin >> l >> r;
        
        long long int sum = 0;
        
        while(l <= r) {
            // std::cout << l - 1 << " : " << arr[l-1] << " - " << r - 1 << " : " << arr[r-1] << std::endl;
          
            if(l == r) {
                sum += (arr[l-1]);
            } else {
              sum += (arr[l-1] + arr[r-1]); 
            }
            
            l++;
            r--;
        }
        
        cout << sum << endl;
      
    }
    
    return 0;
}
