
// https://codeforces.com/group/MWSDmqGsZm/contest/219432/problem/S
// S. Sum of Consecutive Odd Numbers

#include <stdio.h>

int main() {
    int T;
    scanf("%d", &T);
    
    int sum_list[T];
    
    for(int i = 0; i < T; i++)
    {
        int x, y, min, max, sum = 0;
        
        scanf("%d %d", &x, &y);
        
        if(x <= y) 
        {
            min = x;
            max = y;
        }
        else {
            min = y;
            max = x;
        }
        
        int j = min+1;
        
        while(j < max)
        {
            if(j%2 != 0)
            {
                sum += j;
            }
            
            j++;
        }
        
        sum_list[i] = sum;
    }
    
    for(int k = 0; k < T; k++)
    {
        printf("%d\n", sum_list[k]);
    }
    
    

    return 0;
}
